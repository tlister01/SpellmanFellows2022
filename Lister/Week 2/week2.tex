\begin{frame}{Supply and Demand Shocks \footcite{del Rio-Chanona, et al. (2020)}}
	\begin{itemize}
	\item Provided quantitatve predictions of supply and demand shocks due to pandemic
	\item Measured occupations ability to work from home by creating a Remote Labor Index (RLI)
	\item Expected Results due to shocks:
		\begin{itemize}
		\item Would threaten around 20\% of US GDP
		\item Would threaten 23\% of jobs
		\item Reduce total wage income by 16\%
		\end{itemize}
	\item Emphasizes that these results are only first-order shocks and they expect them to be amplified by feedback effects
	\end{itemize}

\end{frame}

\begin{frame}{Looking At Leisure and Hospitality \footcite{Gould and Kassa (2021)}}

	\begin{itemize}
	\item Lesuire and Hospitality lost 8.2 million jobs between March and April 2020 
	\vspace{2mm}
	\item Job shortfalls still remain in Leisure and Hospitality as of Feb 2021
	\vspace{2mm}
	\item Low-wage workers were hit the hardest in Leisure and Hospitality industries
	\end{itemize}
